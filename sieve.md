pair<ll,ll> factor(ll n) {
    ll even=0,odd=0;
    for (ll d : p) {
        if (d*d > n)
            break;
        while (n%d == 0) {
            if(d==2) even++;
            else odd++;
            n /= d;
        }
    }
    if (n>1 && n%2==1) odd++;
    else if(n>1 && n%2==0) even++;
    return mp(even,odd);
}

vi divisors(int num){
	vi d={1};
	while(num>1){
		int spf=lp[num];
		int m=0;
		while(num%spf==0) num/=spf,m++;
		int dz=(int)sz(d);
		int pw=spf;
		for(int i=0;i<m;i++){
			for(int k=0;k<dz;k++) d.eb(d[k]*pw);
			pw*=spf;
		}
	}
	return d;
}```
